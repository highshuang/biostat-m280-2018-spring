function nnmf(
 X::Matrix{T}, 
 r::Int;
 maxiter::Int = 1000, 
 tol::T = 1e-4,
 V::Matrix{T} = rand(T, size(X, 1), r),
 W::Matrix{T} = rand(T, r, size(X, 2))
 )  where T <: AbstractFloat

    # store V*W
    mul_VW = zeros(size(X))
    
    # initialize the denomenator and numerator for V update
    denomenator_V = zeros(size(V))
    numerator_V = zeros(size(V))
    
    # initialize the denomenator and numerator for W update
    denomenator_W = zeros(size(W))
    numerator_W = zeros(size(W))
    
    # initialize the matrix to store r by r matrix
    out_rr = zeros(r, r)
    
    # store the initial norm L
    L = sum(abs2, X - A_mul_B!(mul_VW, V, W))
    
    for t in 1:maxiter # stop after 1000 iterations        
        # update v 
        # denomenator_V = V * W * W'
        # numerator_V = X * W' 
        V .= V .* (A_mul_Bt!(numerator_V, X, W) ./ 
            A_mul_B!(denomenator_V, V, A_mul_Bt!(out_rr, W, W)))
        
        # update W
        # denomenator_W = V' * V * W
        # numerator_W = V' * X 
    
        # update W 
        W .= W .* At_mul_B!(numerator_W, V, X) ./ 
        A_mul_B!(denomenator_W, At_mul_B!(out_rr, V, V), W)  
        
        # check norm difference 
        if abs(sum(abs2, X - A_mul_B!(mul_VW, V, W)) - L) / (L + 1) < tol
            # store norm and quit the loop 
            L = sum(abs2, X - A_mul_B!(mul_VW, V, W))
            break
        end
        
        # update norm 
        L = sum(abs2, X - A_mul_B!(mul_VW, V, W))
        
    end  
    
    # Output
    return L, V, W
end

Question 2 Linear Mixed Models¶

Consider a linear mixed effects model $$ y_i = \mathbf{x}_i^T \beta + \mathbf{z}_i^T \gamma + \epsilon_i, \quad i=1,\ldots,n, $$ where $\epsilon_i$ are independent normal errors $N(0,\sigma_0^2)$, $\beta \in \mathbb{R}^p$ are fixed effects, and $\gamma \in \mathbb{R}^q$ are random effects assumed to be $N(\mathbf{0}_q, \sigma_1^2 \mathbf{I}_q$) independent of $\epsilon_i$.

2.1 Show that $$ \mathbf{y} \sim N \left( \mathbf{X} \beta, \sigma_0^2 \mathbf{I}_n + \sigma_1^2 \mathbf{Z} \mathbf{Z}^T \right), $$ where $\mathbf{y} = (y_1, \ldots, y_n)^T \in \mathbb{R}^n$, $\mathbf{X} = (\mathbf{x}_1, \ldots, \mathbf{x}_n)^T \in \mathbb{R}^{n \times p}$, and $\mathbf{Z} = (\mathbf{z}_1, \ldots, \mathbf{z}_n)^T \in \mathbb{R}^{n \times q}$.

$$ \begin{align} \because \mathbf{y} = \mathbf{X} \beta + \mathbf{Z} \gamma + \epsilon \\ \mathbf{\epsilon} \sim N(\mathbf{0}_n,\sigma_0^2 \mathbf{I}_n) \\ \gamma \sim N(\mathbf{0}_q, \sigma_1^2 \mathbf{I}_q) \\ \epsilon \perp \gamma \\ \end{align} $$

$$ \begin{align} \therefore E(\mathbf{y}) &= E(\mathbf{X} \beta + \mathbf{Z} \gamma + \mathbf{\epsilon})\\ &= E(\mathbf{X} \beta) + E(\mathbf{Z} \gamma) + E(\mathbf{\epsilon}) \\ &= \mathbf{X} \beta + \mathbf{Z} E(\gamma) + \mathbf{0} \\ &= \mathbf{X} \beta + \mathbf{0} + \mathbf{0} \\ &= \mathbf{X} \beta \end{align} $$

$$ \begin{align} Var(\mathbf{y}) &= Var(\mathbf{X} \beta + \mathbf{Z} \gamma + \epsilon) \\ &= Var(\mathbf{X} \beta) + Var(\mathbf{Z} \gamma) + Var(\epsilon) \\ &= \mathbf{0}+ Var(\mathbf{Z} \gamma) + Var(\epsilon) \\ &= \mathbf{Z} Var(\gamma) \mathbf{Z}^T + \sigma_0^2 \mathbf{I}_n\\ &= \mathbf{Z} \sigma_1^2 \mathbf{I}_q \mathbf{Z}^T + \sigma_0^2 \mathbf{I}_n\\ &= \sigma_1^2 \mathbf{Z} \mathbf{Z}^T + \sigma_0^2 \mathbf{I}_n \end{align} $$

Therefore $\mathbf{y} \sim N \left( \mathbf{X} \beta, \sigma_0^2 \mathbf{I}_n + \sigma_1^2 \mathbf{Z} \mathbf{Z}^T \right)$

  1. Write a function, with interface 
    logpdf_mvn(y::Vector, Z::Matrix, 0::Number, 1::Number),
    that evaluates the log-density of a multivariate normal with mean $\mathbf{0}$ and covariance $\sigma_0^2 \mathbf{I} + \sigma_1^2 \mathbf{Z} \mathbf{Z}^T$ at $\mathbf{y}$. Make your code efficient in the $n \gg q$ case.

With determinant identity and Woodnury formula, $$ \begin{align} det\Sigma =& det(\sigma_0^2 \mathbf{I} + \sigma_1^2 \mathbf{Z} \mathbf{Z}^T)\\ = & det(\sigma_0^2 \mathbf{I}) *det(\mathbf{I}+\sigma_1^2\mathbf{Z}^T\mathbf{I}^{-1}\mathbf{Z})\\ = & \sigma_0^{2n} * det((\mathbf{I}+\frac{\sigma_1^2}{\sigma_0^2}\mathbf{Z}^T\mathbf{Z}) \end{align} $$

$$ \begin{align} \Sigma^{-1} =& (\sigma_0^2 \mathbf{I} + \sigma_1^2 \mathbf{Z} \mathbf{Z}^T)^{-1}\\ =& (\sigma_0^2 \mathbf{I})^{-1}-(\sigma_0^2 \mathbf{I})^{-1}\sigma_1^2\mathbf{Z}(I+\mathbf{Z}^T(\sigma_0^2 \mathbf{I})^{-1}\mathbf{Z})^{-1}\mathbf{Z}^T(\sigma_0^2 \mathbf{I})^{-1}\\ =&\frac{1}{\sigma_0^2}\mathbf{I}-\frac{\sigma_1^2}{\sigma_0^4}\mathbf{Z}(I+\frac{\sigma_1^2}{\sigma_0^2}\mathbf{Z}^T\mathbf{Z})^{-1}\mathbf{Z}^T \end{align} $$

function logpdf_mvn_3(
 y::Vector,
 Z::Matrix,
 0::Number,
 1::Number
) 
    n = length(y)
    # using Cholesky decomposition  
    chol = cholfact(Symmetric(I + 1^2 / 0^2 * Z'Z))
    
    # logdet
    logdet = 2 * n * log(0) + logdet(chol)

    # log density
    - (n / 2) * log(2) - (1 / 2) * logdet - 1 / (2 * 0^2) * dot(y, y) + 
    1/2 * sum(abs2, chol[:L]\((Z'y) * (1 / 0^2)))
end
